# Zuul网关

# 配置文件

server:
  port: 80
spring:
  application:
    name: zuul

eureka:
  instance:
    hostname: localhost
    prefer-ip-address: true
  client:
    serviceUrl:
      defaultZone: http://${eureka.instance.hostname}:8761/eureka/
zuul:
  routes:
    util:
      path: /hello/**
      serviceId: EUREKACLIENT
    search:
      path: /**
      url: https://cn.bing.com/
